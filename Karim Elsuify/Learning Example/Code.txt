

void setup() {
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(6, OUTPUT);
  pinMode(7, OUTPUT);
}

void loop() {

  for (int i = 4; i <= 7; i++)
  {
    digitalWrite(i, HIGH);
    delay(200);

    digitalWrite(i, LOW);
  }
  for (int i = 7; i >= 4; i--)
  {
    digitalWrite(i, HIGH);
    delay(200);

    digitalWrite(i, LOW);
  }
}
