/*
  The system produce a pulse of duration
  equal to the supplied 8 bit value (0-255) multiplied by 4.
*/

void setup() {
  pinMode(13,OUTPUT);
for(int i=0; i<8;i++) 
 pinMode(i,INPUT);
}

void loop() {
  int decimal = 0;
  
  for(int i=0; i<8;i++) 
      if (digitalRead(i) == HIGH)
        decimal += pow(2,i);
        
   decimal *= 4;

   digitalWrite(13,HIGH);
   delay(decimal);
   digitalWrite(13,LOW);
   delay(500);
}
